drop table transferencias;
drop table times;
drop table jogadores;
drop table torneios;
drop table partidas;
drop table eventos;
 
create table times(
	id serial primary key,
	nome text not null,
	localidade text
);

create table transferencias(
	id serial primary key,
	time_origem integer not null,
	time_destino integer not null,
	id_jogador integer not null,
	data date,
	valor real
);

create table jogadores(
	id serial primary key,
	nome text not null,
	data_de_nascimento date not null,
	pais text not null,
	time integer
);

create table torneios(
	id_torneio serial primary key,
	nome_torneio text not null
);

create table partidas(
	id_partida serial primary key,
	id_torneio integer not null,
	id_time1 integer not null,
	id_time2 integer not null
);

create table eventos(
	id_evento serial primary key,
	id_partida integer,
	descricao_evento text not null
);
